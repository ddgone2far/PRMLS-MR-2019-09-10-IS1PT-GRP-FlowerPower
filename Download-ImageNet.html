def download_multiple_times(url, full_name, counter):
    print("Try %s" %(counter))
    try:
        #wget.download(url, full_name)
        request = urllib.request.urlopen(url, timeout=10)
        with open(full_name, 'wb') as f:
            try:
                f.write(request.read())
            except:
                print("error")
        check_img_or_del(full_name)
        print("Downloaded successfully")
        return 0
    except:
        counter_new = counter - 1
        if counter == 1:
            print("Could not find the image %s" %(url))
        else:
            return download_multiple_times(url, full_name, counter_new)
